#include<iostream>
using namespace std;

class classroom{
    private:
    string player;
    int health;
    float rating;
    public:
    //default constructor is required if we are using the parameterised constructor 
    classroom(){ 
        player = "Demon";
        health =120;
        rating =9.8;
    }
    classroom(int x){
        player ="five";
        health =x;
        rating =9.8;
    }
    
    ~classroom(){
        cout<<"Destructor is called"<<endl;
    }
    
    void accept(){
        cout<<"Enter character name : ";
        getline(cin,player);
        cout<<"Enter The Player health : ";
        cin>>health;
        cout<<"Enter The Rating of the player : ";
        cin>>rating;
    }
    
    void display(){
        if(player!=""){
            cout<<player<<"\t"<<health<<"\t"<<rating<<endl;
        }
    }
    
};
